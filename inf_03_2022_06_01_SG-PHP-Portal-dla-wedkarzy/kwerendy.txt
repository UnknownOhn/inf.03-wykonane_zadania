SELECT ryby.id, ryby.nazwa, ryby.wystepowanie
FROM ryby
WHERE ryby.styl_zycia = 1;

-- zapytanie 2
SELECT Ryby_id, wymiar_ochronny 
FROM okres_ochronny 
WHERE wymiar_ochronny < 30; 

-- zapytanie 3
SELECT ryby.nazwa, lowisko.akwen, lowisko.wojewodztwo
FROM lowisko
JOIN ryby ON ryby.id = lowisko.Ryby_id
WHERE lowisko.rodzaj = 3;
