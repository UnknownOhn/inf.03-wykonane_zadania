Zapytanie 1
SELECT cena, pojemnosc
FROM farby
WHERE kolor = "czerwony"

Zapytanie 2
SELECT kolor, cena/pojemnosc AS "Cena jednostkowa"
FROM farby;

Zapytanie 3
SELECT kolor, liczba_puszek
FROM farby
JOIN malowanie ON malowanie.id_farby = farby.id_farby
WHERE id_pomieszczenia = 3;

Zapytanie 4
CREATE VIEW id_pomieszczenie1 AS SELECT id_sciany, id_farby, liczba_puszek
FROM malowanie
WHERE id_pomieszczenia = 1;
