SELECT uslugi.cena, uslugi.nazwa
FROM uslugi
WHERE uslugi.cena >= 50;

-- 2
SELECT AVG(uslugi.cena) AS 'Średnia cena arytmetyczna', COUNT(*) AS 'Liczba usług'
FROM uslugi
GROUP BY uslugi.rodzaj = "1";
